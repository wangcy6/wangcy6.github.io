https://github.com/bitcoin/bitcoin/blob/master/doc/build-unix.md

yum install autoconf automake libtool libdb-devel boost-devel libevent-devel -y
yum install -y  openssl-devel gcc-c++

yum install  -y  sqlite-devel
#checking whether to build wallet with support for sqlite... no
yum install  -y  qt5-qtquickcontrols qt5-qtdeclarative-devel -y
# configure: WARNING: Qt5Core >= 5.11.3 not found; bitcoin-qt frontend will not be built

yum install -y epel-release
yum install -y zeromq-devel

## libdb4.8	Berkeley DB	Wallet storage (only needed when legacy wallet enabled)


git git@github.com:wangcy6/bitcoin.git
https://github.com/bitcoin/bitcoin/blob/master/doc/dependencies.md

cd /root/bitcoin
./autogen.sh
./configure

./configure  --enable-cxx --disable-shared --with-pic --without-miniupnpc --without-natpmp --without-gui

make
